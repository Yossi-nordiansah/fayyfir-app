---

📘 TABEL MASTER

CREATE TABLE bahan_master (
  id INT AUTO_INCREMENT PRIMARY KEY,
  nama_bahan VARCHAR(100) NOT NULL,
  satuan VARCHAR(20) NOT NULL,
  keterangan TEXT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE supplier (
  id INT AUTO_INCREMENT PRIMARY KEY,
  nama_supplier VARCHAR(100) NOT NULL,
  kontak VARCHAR(50),
  alamat TEXT,
  catatan TEXT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE buyer (
  id INT AUTO_INCREMENT PRIMARY KEY,
  nama_buyer VARCHAR(100) NOT NULL,
  kontak VARCHAR(50),
  alamat TEXT,
  catatan TEXT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


---

📘 TABEL PEMBELIAN AWAL

CREATE TABLE pembelian_awal (
  id INT AUTO_INCREMENT PRIMARY KEY,
  kode_batch VARCHAR(30) UNIQUE,
  tanggal_pembelian DATE NOT NULL,
  id_bahan INT,
  id_supplier INT,
  berat_awal DECIMAL(10,2),
  harga_per_kg DECIMAL(12,2),
  total_modal DECIMAL(14,2),
  catatan TEXT,
  status VARCHAR(20) DEFAULT 'proses',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (id_bahan) REFERENCES bahan_master(id),
  FOREIGN KEY (id_supplier) REFERENCES supplier(id)
);


---

📘 TABEL PROSES PRODUKSI

CREATE TABLE proses_jemur (
  id INT AUTO_INCREMENT PRIMARY KEY,
  id_pembelian INT,
  tanggal_mulai DATE,
  tanggal_selesai DATE,
  berat_setelah_jemur DECIMAL(10,2),
  penyusutan_jemur DECIMAL(5,2),
  keterangan TEXT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (id_pembelian) REFERENCES pembelian_awal(id)
);

CREATE TABLE proses_kupas (
  id INT AUTO_INCREMENT PRIMARY KEY,
  id_pembelian INT,
  berat_setelah_kupas DECIMAL(10,2),
  penyusutan_kupas DECIMAL(5,2),
  tanggal_proses DATE,
  keterangan TEXT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (id_pembelian) REFERENCES pembelian_awal(id)
);

CREATE TABLE proses_sortir (
  id INT AUTO_INCREMENT PRIMARY KEY,
  id_pembelian INT,
  berat_akhir DECIMAL(10,2),
  penyusutan_total DECIMAL(5,2),
  tanggal_simpan DATE,
  lokasi_simpan VARCHAR(100),
  catatan TEXT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (id_pembelian) REFERENCES pembelian_awal(id)
);


---

📘 TABEL PENJUALAN

CREATE TABLE penjualan (
  id INT AUTO_INCREMENT PRIMARY KEY,
  id_pembelian INT,
  id_buyer INT,
  tanggal_jual DATE,
  berat_jual DECIMAL(10,2),
  harga_jual_per_kg DECIMAL(12,2),
  total_penjualan DECIMAL(14,2),
  laba_bersih DECIMAL(14,2),
  keterangan TEXT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (id_pembelian) REFERENCES pembelian_awal(id),
  FOREIGN KEY (id_buyer) REFERENCES buyer(id)
);

📘 VIEW & LAPORAN OTOMATIS (opsional)

1. View Ringkasan Modal & HPP

CREATE VIEW v_hpp_awal AS
SELECT
  p.id,
  b.nama_bahan,
  s.nama_supplier,
  p.tanggal_pembelian,
  p.berat_awal,
  p.harga_per_kg,
  p.total_modal,
  (p.total_modal / NULLIF(p.berat_awal, 0)) AS hpp_per_kg
FROM pembelian_awal p
JOIN bahan_master b ON p.id_bahan = b.id
JOIN supplier s ON p.id_supplier = s.id;


2. View Penyusutan per Tahap

CREATE VIEW v_penyusutan_tahap AS
SELECT
  pa.id AS id_pembelian,
  pa.kode_batch,
  pj.penyusutan_jemur,
  pk.penyusutan_kupas,
  ps.penyusutan_total
FROM pembelian_awal pa
LEFT JOIN proses_jemur pj ON pa.id = pj.id_pembelian
LEFT JOIN proses_kupas pk ON pa.id = pk.id_pembelian
LEFT JOIN proses_sortir ps ON pa.id = ps.id_pembelian;
